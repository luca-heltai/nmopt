190template <int dim>
191void
192InversePoissonKKT<dim>::setup_system()
193{
194 dof_handler.distribute_dofs(*fe);
195
196 const std::vector<unsigned int> block_component = {state_block,
197 adjoint_block,
198 control_block};
199 DoFRenumbering::component_wise(dof_handler, block_component);
200
201 dofs_per_block =
202 DoFTools::count_dofs_per_fe_block(dof_handler, block_component);
203
204 constraints.clear();
205 DoFTools::make_hanging_node_constraints(dof_handler, constraints);
206 VectorTools::interpolate_boundary_values(
207 dof_handler,
208 0,
209 Functions::ZeroFunction<dim>(n_blocks),
210 constraints,
211 fe->component_mask(FEValuesExtractors::Scalar(state_block)));
212 VectorTools::interpolate_boundary_values(
213 dof_handler,
214 0,
215 Functions::ZeroFunction<dim>(n_blocks),
216 constraints,
217 fe->component_mask(FEValuesExtractors::Scalar(adjoint_block)));
218 constraints.close();
219
220 BlockDynamicSparsityPattern dsp(dofs_per_block, dofs_per_block);
221 DoFTools::make_sparsity_pattern(dof_handler,
222 dsp,
223 constraints,
224 /*keep_constrained_dofs=*/false);
225 sparsity_pattern.copy_from(dsp);
226
227 system_matrix.reinit(sparsity_pattern);
228}

280template <int dim>
281void
282InversePoissonKKT<dim>::initialize_control_mass_matrix()
283{
284 AssertThrow(regularization > 0.0,
285 ExcMessage("Regularization must be strictly positive."));
286
287 DynamicSparsityPattern dsp(dofs_per_block[control_block],
288 dofs_per_block[control_block]);
289 std::vector<types::global_dof_index> local_dof_indices(fe->n_dofs_per_cell());
290 const auto offset = control_global_offset();
291
292 for (const auto &cell : dof_handler.active_cell_iterators())
293 {
294 cell->get_dof_indices(local_dof_indices);
295
296 for (unsigned int i = 0; i < fe->n_dofs_per_cell(); ++i)
297 if (fe->system_to_block_index(i).first == control_block)
298 for (unsigned int j = 0; j < fe->n_dofs_per_cell(); ++j)
299 if (fe->system_to_block_index(j).first == control_block)
300 dsp.add(local_dof_indices[i] - offset,
301 local_dof_indices[j] - offset);
302 }
303
304 SparsityPattern sparsity;
305 sparsity.copy_from(dsp);
306 control_mass_matrix.reinit(sparsity);
307
308 const QGauss<dim> quadrature_formula(std::max(state_degree, control_degree) +
309 1);
310 FEValues<dim> fe_values(*fe,
311 quadrature_formula,
312 update_values | update_JxW_values);
313 const FEValuesExtractors::Scalar control(control_block);
314
315 FullMatrix<double> cell_matrix(fe->n_dofs_per_cell(), fe->n_dofs_per_cell());
316
317 for (const auto &cell : dof_handler.active_cell_iterators())
318 {
319 cell_matrix = 0;
320 fe_values.reinit(cell);
321
322 for (const unsigned int q_index : fe_values.quadrature_point_indices())
323 for (const unsigned int i : fe_values.dof_indices())
324 {
325 const double phi_control_i = fe_values[control].value(i, q_index);
326
327 if (phi_control_i == 0.0)
328 continue;
329
330 for (const unsigned int j : fe_values.dof_indices())
331 {
332 const double phi_control_j =
333 fe_values[control].value(j, q_index);
334
335 if (phi_control_j == 0.0)
336 continue;
337
338 cell_matrix(i, j) +=
339 phi_control_i * phi_control_j * fe_values.JxW(q_index);
340 }
341 }
342
343 cell->get_dof_indices(local_dof_indices);
344 for (const unsigned int i : fe_values.dof_indices())
345 if (fe->system_to_block_index(i).first == control_block)
346 for (const unsigned int j : fe_values.dof_indices())
347 if (fe->system_to_block_index(j).first == control_block)
348 control_mass_matrix.add(local_dof_indices[i] - offset,
349 local_dof_indices[j] - offset,
350 cell_matrix(i, j));
351 }
352
353 control_mass_inverse.initialize(control_mass_matrix);
354}

370template <int dim>
371void
372InversePoissonKKT<dim>::assemble_system()
373{
374 system_matrix = 0;
375 system_rhs = 0;
376 residual = 0;
377
378 const QGauss<dim> quadrature_formula(std::max(state_degree, control_degree) +
379 1);
380 FEValues<dim> fe_values(*fe,
381 quadrature_formula,
382 update_values | update_gradients |
383 update_quadrature_points | update_JxW_values);
384
385 const FEValuesExtractors::Scalar state(state_block);
386 const FEValuesExtractors::Scalar adjoint(adjoint_block);
387 const FEValuesExtractors::Scalar control(control_block);
388
389 const unsigned int dofs_per_cell = fe->n_dofs_per_cell();
390
391 FullMatrix<double> cell_matrix(dofs_per_cell, dofs_per_cell);
392 Vector<double> cell_residual(dofs_per_cell);
393
394 std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
395
396 std::vector<double> state_values(quadrature_formula.size());
397 std::vector<Tensor<1, dim>> state_gradients(quadrature_formula.size());
398 std::vector<double> adjoint_values(quadrature_formula.size());
399 std::vector<Tensor<1, dim>> adjoint_gradients(quadrature_formula.size());
400 std::vector<double> control_values(quadrature_formula.size());
401
402 for (const auto &cell : dof_handler.active_cell_iterators())
403 {
404 cell_matrix = 0;
405 cell_residual = 0;
406
407 fe_values.reinit(cell);
408 fe_values[state].get_function_values(current_solution, state_values);
409 fe_values[state].get_function_gradients(current_solution,
410 state_gradients);
411 fe_values[adjoint].get_function_values(current_solution, adjoint_values);
412 fe_values[adjoint].get_function_gradients(current_solution,
413 adjoint_gradients);
414 fe_values[control].get_function_values(current_solution, control_values);
415
416 for (const unsigned int q_index : fe_values.quadrature_point_indices())
417 {
418 const Point<dim> &x_q = fe_values.quadrature_point(q_index);
419 const double forcing_value = forcing_term.value(x_q);
420 const double desired_value = desired_state_function.value(x_q);
421 const double reference_value =
422 reference_coefficient_function.value(x_q);
423
424 const double coefficient_value = control_values[q_index];
425 const double state_value = state_values[q_index];
426 const double adjoint_value = adjoint_values[q_index];
427 const Tensor<1, dim> &state_gradient = state_gradients[q_index];
428 const Tensor<1, dim> &adjoint_gradient = adjoint_gradients[q_index];
429
430 (void)adjoint_value;
431
432 for (const unsigned int i : fe_values.dof_indices())
433 {
434 const double phi_state_i = fe_values[state].value(i, q_index);
435 const Tensor<1, dim> grad_state_i =
436 fe_values[state].gradient(i, q_index);
437 const double phi_adjoint_i = fe_values[adjoint].value(i, q_index);
438 const Tensor<1, dim> grad_adjoint_i =
439 fe_values[adjoint].gradient(i, q_index);
440 const double phi_control_i = fe_values[control].value(i, q_index);
441
442 cell_residual(i) +=
443 (coefficient_value * (state_gradient * grad_state_i) -
444 forcing_value * phi_state_i +
445 coefficient_value * (adjoint_gradient * grad_adjoint_i) +
446 (state_value - desired_value) * phi_adjoint_i +
447 (regularization * (coefficient_value - reference_value) -
448 state_gradient * adjoint_gradient) *
449 phi_control_i) *
450 fe_values.JxW(q_index);
451
452 for (const unsigned int j : fe_values.dof_indices())
453 {
454 const double phi_state_j = fe_values[state].value(j, q_index);
455 const Tensor<1, dim> grad_state_j =
456 fe_values[state].gradient(j, q_index);
457 const double phi_adjoint_j =
458 fe_values[adjoint].value(j, q_index);
459 const Tensor<1, dim> grad_adjoint_j =
460 fe_values[adjoint].gradient(j, q_index);
461 const double phi_control_j =
462 fe_values[control].value(j, q_index);
463
464 cell_matrix(i, j) +=
465 (coefficient_value * (grad_state_j * grad_state_i) +
466 phi_control_j * (state_gradient * grad_state_i) +
467 phi_state_j * phi_adjoint_i +
468 coefficient_value * (grad_adjoint_j * grad_adjoint_i) +
469 phi_control_j * (adjoint_gradient * grad_adjoint_i) +
470 regularization * phi_control_j * phi_control_i -
471 phi_control_i * (grad_state_j * adjoint_gradient) -
472 phi_control_i * (state_gradient * grad_adjoint_j)) *
473 fe_values.JxW(q_index);
474
475 (void)phi_adjoint_j;
476 }
477 }
478 }
479
480 cell->get_dof_indices(local_dof_indices);
481
482 constraints.distribute_local_to_global(
483 cell_matrix, cell_residual, local_dof_indices, system_matrix, residual);
484 }
485
486 system_rhs = residual;
487 system_rhs *= -1.0;
488}

492template <int dim>
493BlockVector<double>
494InversePoissonKKT<dim>::assemble_residual_only(
495 const BlockVector<double> &iterate) const
496{
497 BlockVector<double> local_residual(iterate);
498 local_residual = 0;
499
500 const QGauss<dim> quadrature_formula(std::max(state_degree, control_degree) +
501 1);
502 FEValues<dim> fe_values(*fe,
503 quadrature_formula,
504 update_values | update_gradients |
505 update_quadrature_points | update_JxW_values);
506
507 const FEValuesExtractors::Scalar state(state_block);
508 const FEValuesExtractors::Scalar adjoint(adjoint_block);
509 const FEValuesExtractors::Scalar control(control_block);
510
511 Vector<double> cell_residual(fe->n_dofs_per_cell());
512 std::vector<types::global_dof_index> local_dof_indices(fe->n_dofs_per_cell());
513
514 std::vector<double> state_values(quadrature_formula.size());
515 std::vector<Tensor<1, dim>> state_gradients(quadrature_formula.size());
516 std::vector<double> adjoint_values(quadrature_formula.size());
517 std::vector<Tensor<1, dim>> adjoint_gradients(quadrature_formula.size());
518 std::vector<double> control_values(quadrature_formula.size());
519
520 for (const auto &cell : dof_handler.active_cell_iterators())
521 {
522 cell_residual = 0;
523
524 fe_values.reinit(cell);
525 fe_values[state].get_function_values(iterate, state_values);
526 fe_values[state].get_function_gradients(iterate, state_gradients);
527 fe_values[adjoint].get_function_values(iterate, adjoint_values);
528 fe_values[adjoint].get_function_gradients(iterate, adjoint_gradients);
529 fe_values[control].get_function_values(iterate, control_values);
530
531 for (const unsigned int q_index : fe_values.quadrature_point_indices())
532 {
533 const Point<dim> &x_q = fe_values.quadrature_point(q_index);
534 const double forcing_value = forcing_term.value(x_q);
535 const double desired_value = desired_state_function.value(x_q);
536 const double reference_value =
537 reference_coefficient_function.value(x_q);
538
539 const double coefficient_value = control_values[q_index];
540 const double state_value = state_values[q_index];
541 const Tensor<1, dim> &state_gradient = state_gradients[q_index];
542 const Tensor<1, dim> &adjoint_gradient = adjoint_gradients[q_index];
543
544 (void)adjoint_values;
545
546 for (const unsigned int i : fe_values.dof_indices())
547 {
548 const double phi_state_i = fe_values[state].value(i, q_index);
549 const Tensor<1, dim> grad_state_i =
550 fe_values[state].gradient(i, q_index);
551 const double phi_adjoint_i = fe_values[adjoint].value(i, q_index);
552 const Tensor<1, dim> grad_adjoint_i =
553 fe_values[adjoint].gradient(i, q_index);
554 const double phi_control_i = fe_values[control].value(i, q_index);
555
556 cell_residual(i) +=
557 (coefficient_value * (state_gradient * grad_state_i) -
558 forcing_value * phi_state_i +
559 coefficient_value * (adjoint_gradient * grad_adjoint_i) +
560 (state_value - desired_value) * phi_adjoint_i +
561 (regularization * (coefficient_value - reference_value) -
562 state_gradient * adjoint_gradient) *
563 phi_control_i) *
564 fe_values.JxW(q_index);
565 }
566 }
567
568 cell->get_dof_indices(local_dof_indices);
569 constraints.distribute_local_to_global(cell_residual,
570 local_dof_indices,
571 local_residual);
572 }
573
574 return local_residual;
575}

579template <int dim>
580Vector<double>
581InversePoissonKKT<dim>::compute_stationarity() const
582{
583 Vector<double> dual_stationarity(dofs_per_block[control_block]);
584 dual_stationarity = residual.block(control_block);
585
586 Vector<double> stationarity(dofs_per_block[control_block]);
587 control_mass_inverse.vmult(stationarity, dual_stationarity);
588
589 return stationarity;
590}

594template <int dim>
595std::vector<bool>
596InversePoissonKKT<dim>::compute_lower_active_set() const
597{
598 const auto &control_lower = lower_bound.block(control_block);
599 const auto &control_value = current_solution.block(control_block);
600 const auto stationarity = compute_stationarity();
601
602 std::vector<bool> lower_active(control_value.size(), false);
603
604 for (unsigned int i = 0; i < control_value.size(); ++i)
605 {
606 const double indicator =
607 stationarity[i] +
608 active_set_parameter * (control_value[i] - control_lower[i]);
609 if (indicator < -newton_tolerance)
610 lower_active[i] = true;
611 }
612
613 return lower_active;
614}

618template <int dim>
619std::vector<bool>
620InversePoissonKKT<dim>::compute_upper_active_set() const
621{
622 const auto &control_upper = upper_bound.block(control_block);
623 const auto &control_value = current_solution.block(control_block);
624 const auto stationarity = compute_stationarity();
625
626 std::vector<bool> upper_active(control_value.size(), false);
627
628 for (unsigned int i = 0; i < control_value.size(); ++i)
629 {
630 const double indicator =
631 stationarity[i] +
632 active_set_parameter * (control_value[i] - control_upper[i]);
633 if (indicator > newton_tolerance)
634 upper_active[i] = true;
635 }
636
637 return upper_active;
638}

642template <int dim>
643AffineConstraints<double>
644InversePoissonKKT<dim>::make_control_constraints(
645 const std::vector<bool> &lower_active,
646 const std::vector<bool> &upper_active) const
647{
648 AssertDimension(lower_active.size(), upper_active.size());
649
650 AffineConstraints<double> control_constraints;
651 const auto offset = control_global_offset();
652
653 for (unsigned int i = 0; i < lower_active.size(); ++i)
654 if (lower_active[i] || upper_active[i])
655 {
656 control_constraints.add_line(offset + i);
657 control_constraints.set_inhomogeneity(
658 offset + i,
659 lower_active[i] ? lower_bound.block(control_block)[i] -
660 current_solution.block(control_block)[i] :
661 upper_bound.block(control_block)[i] -
662 current_solution.block(control_block)[i]);
663 }
664
665 control_constraints.close();
666
667 AffineConstraints<double> combined_constraints;
668 combined_constraints.copy_from(constraints);
669 combined_constraints.merge(control_constraints);
670 return combined_constraints;
671}

675template <int dim>
676void
677InversePoissonKKT<dim>::solve_linearized_system(
678 const std::vector<bool> &lower_active,
679 const std::vector<bool> &upper_active,
680 BlockVector<double> &update) const
681{
682 const AffineConstraints<double> combined_constraints =
683 make_control_constraints(lower_active, upper_active);
684
685 BlockSparseMatrix<double> modified_matrix;
686 modified_matrix.reinit(sparsity_pattern);
687 modified_matrix.copy_from(system_matrix);
688
689 BlockVector<double> modified_rhs(system_rhs);
690
691 combined_constraints.condense(modified_matrix, modified_rhs);
692
693 SparseDirectUMFPACK direct_solver;
694 direct_solver.initialize(modified_matrix);
695 direct_solver.vmult(update, modified_rhs);
696 combined_constraints.distribute(update);
697}

702void
703InversePoissonKKT<dim>::enforce_box_constraints(
704 BlockVector<double> &iterate,
705 const std::vector<bool> &lower_active,
706 const std::vector<bool> &upper_active) const
707{
708 constraints.distribute(iterate);
709
710 for (unsigned int i = 0; i < lower_active.size(); ++i)
711 if (lower_active[i])
712 iterate.block(control_block)[i] = lower_bound.block(control_block)[i];
713 else if (upper_active[i])
714 iterate.block(control_block)[i] = upper_bound.block(control_block)[i];
715
716 for (unsigned int i = 0; i < lower_active.size(); ++i)
717 if (!lower_active[i] && !upper_active[i])
718 iterate.block(control_block)[i] =
719 std::clamp(iterate.block(control_block)[i],
720 lower_bound.block(control_block)[i],
721 upper_bound.block(control_block)[i]);
722}

830template <int dim>
831void
832InversePoissonKKT<dim>::run()
833{
834 make_grid();
835 setup_system();
836 initialize_vectors();
837
838 interpolate_component_function(lower_bound_function,
839 control_block,
840 lower_bound);
841 interpolate_component_function(upper_bound_function,
842 control_block,
843 upper_bound);
844 interpolate_component_function(reference_coefficient_function,
845 control_block,
846 reference_coefficient_vector);
847 interpolate_component_function(desired_state_function,
848 state_block,
849 desired_state_vector);
850 interpolate_component_function(exact_coefficient_function,
851 control_block,
852 exact_coefficient_vector);
853
854 const auto &control_lower = lower_bound.block(control_block);
855 const auto &control_upper = upper_bound.block(control_block);
856 for (unsigned int i = 0; i < control_lower.size(); ++i)
857 AssertThrow(control_lower[i] <= control_upper[i],
858 ExcMessage("Lower bound exceeds upper bound."));
859
860 initialize_solution();
861 enforce_box_constraints(current_solution,
862 std::vector<bool>(control_lower.size(), false),
863 std::vector<bool>(control_lower.size(), false));
864 initialize_control_mass_matrix();
865
866 std::vector<bool> lower_active(control_lower.size(), false);
867 std::vector<bool> upper_active(control_lower.size(), false);
868 std::vector<std::pair<double, std::string>> times_and_names;
869
870 for (unsigned int iteration = 0; iteration < max_pdas_iterations; ++iteration)
871 {
872 assemble_system();
873 const double current_residual_norm = residual.l2_norm();
874
875 const auto next_lower_active = compute_lower_active_set();
876 const auto next_upper_active = compute_upper_active_set();
877
878 BlockVector<double> newton_update(current_solution);
879 newton_update = 0;
880 solve_linearized_system(next_lower_active,
881 next_upper_active,
882 newton_update);
883
884 BlockVector<double> next_solution(current_solution);
885 BlockVector<double> next_residual(current_solution);
886 next_residual = residual;
887 double accepted_step = 0.0;
888 double residual_norm = std::numeric_limits<double>::infinity();
889 double best_norm = current_residual_norm;
890 double best_step = 0.0;
891 auto best_solution = current_solution;
892 auto best_residual = residual;
893
894 for (double step_length = 1.0; step_length >= 1.0 / 1024.0;
895 step_length *= 0.5)
896 {
897 BlockVector<double> trial_solution(current_solution);
898 trial_solution.add(step_length, newton_update);
899 enforce_box_constraints(trial_solution,
900 next_lower_active,
901 next_upper_active);
902
903 const BlockVector<double> trial_residual =
904 assemble_residual_only(trial_solution);
905 const double trial_norm = trial_residual.l2_norm();
906
907 if (trial_norm < best_norm)
908 {
909 best_norm = trial_norm;
910 best_step = step_length;
911 best_solution = trial_solution;
912 best_residual = trial_residual;
913 }
914
915 if (trial_norm < current_residual_norm || trial_norm < 1e-12)
916 {
917 next_solution = trial_solution;
918 next_residual = trial_residual;
919 residual_norm = trial_norm;
920 accepted_step = step_length;
921 break;
922 }
923 }
924
925 if (accepted_step == 0.0)
926 {
927 next_solution = best_solution;
928 next_residual = best_residual;
929 residual_norm = best_norm;
930 accepted_step = best_step;
931 }
932
933 BlockVector<double> update(next_solution);
934 update -= current_solution;
935 const double update_norm = update.l2_norm();
936
937 const unsigned int n_lower =
938 std::count(next_lower_active.begin(), next_lower_active.end(), true);
939 const unsigned int n_upper =
940 std::count(next_upper_active.begin(), next_upper_active.end(), true);
941
942 std::cout << "PDAS/Newton iteration " << iteration
943 << ": |A_-|=" << n_lower << ", |A_+|=" << n_upper
944 << ", step=" << accepted_step << ", ||delta||=" << update_norm
945 << ", ||R||=" << residual_norm << std::endl;
946
947 times_and_names.emplace_back(static_cast<double>(iteration),
948 output_iteration(iteration,
949 next_solution,
950 next_lower_active,
951 next_upper_active));
952
953 if (next_lower_active == lower_active &&
954 next_upper_active == upper_active && update_norm < newton_tolerance &&
955 residual_norm < newton_tolerance)
956 {
957 current_solution = next_solution;
958 residual = next_residual;
959 std::cout << "Converged after " << iteration + 1
960 << " PDAS/Newton iterations." << std::endl;
961 break;
962 }
963
964 if (next_lower_active == lower_active &&
965 next_upper_active == upper_active && accepted_step == 0.0)
966 {
967 current_solution = next_solution;
968 residual = next_residual;
969 std::cout << "Stopped because the damped Newton step no longer"
970 << " improves the KKT residual." << std::endl;
971 break;
972 }
973
974 current_solution = next_solution;
975 residual = next_residual;
976 lower_active = next_lower_active;
977 upper_active = next_upper_active;
978
979 if (iteration + 1 == max_pdas_iterations)
980 std::cout << "Reached the maximum number of PDAS iterations."
981 << std::endl;
982 }
983
984 output_results(current_solution, lower_active, upper_active, output_name);
985
986 std::ofstream pvd_output(output_name + ".pvd");
987 DataOutBase::write_pvd_record(pvd_output, times_and_names);
988}
