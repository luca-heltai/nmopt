196template <int dim>
197void
198KKT<dim>::setup_system()
199{
200 dof_handler.distribute_dofs(*fe);
201
202 const std::vector<unsigned int> block_component = {state_block,
203 adjoint_block,
204 control_block};
205 DoFRenumbering::component_wise(dof_handler, block_component);
206
207 dofs_per_block =
208 DoFTools::count_dofs_per_fe_block(dof_handler, block_component);
209
210 constraints.clear();
211 DoFTools::make_hanging_node_constraints(dof_handler, constraints);
212 const VectorFunctionFromScalarFunctionObject<dim> state_bc(
213 [this](const Point<dim> &p) { return state_boundary_values.value(p); },
214 state_block,
215 n_blocks);
216 const VectorFunctionFromScalarFunctionObject<dim> adjoint_bc(
217 [this](const Point<dim> &p) { return adjoint_boundary_values.value(p); },
218 adjoint_block,
219 n_blocks);
220 VectorTools::interpolate_boundary_values(
221 dof_handler,
222 0,
223 state_bc,
224 constraints,
225 fe->component_mask(FEValuesExtractors::Scalar(state_block)));
226 VectorTools::interpolate_boundary_values(
227 dof_handler,
228 0,
229 adjoint_bc,
230 constraints,
231 fe->component_mask(FEValuesExtractors::Scalar(adjoint_block)));
232 constraints.close();
233
234 BlockDynamicSparsityPattern dsp(dofs_per_block, dofs_per_block);
235
236 DoFTools::make_sparsity_pattern(dof_handler,
237 dsp,
238 constraints,
239 /*keep_constrained_dofs = */ false);
240 sparsity_pattern.copy_from(dsp);
241
242 system_matrix.reinit(sparsity_pattern);
243
244 solution.reinit(std::vector<types::global_dof_index>(dofs_per_block.begin(),
245 dofs_per_block.end()));
246 system_rhs.reinit(std::vector<types::global_dof_index>(dofs_per_block.begin(),
247 dofs_per_block.end()));
248}

252template <int dim>
253void
254KKT<dim>::assemble_system()
255{
256 system_matrix = 0;
257 system_rhs = 0;
258
259 const QGauss<dim> quadrature_formula(std::max(state_degree, control_degree) +
260 1);
261 FEValues<dim> fe_values(*fe,
262 quadrature_formula,
263 update_values | update_gradients |
264 update_quadrature_points | update_JxW_values);
265
266 const FEValuesExtractors::Scalar state(state_block);
267 const FEValuesExtractors::Scalar adjoint(adjoint_block);
268 const FEValuesExtractors::Scalar control(control_block);
269
270 const unsigned int dofs_per_cell = fe->n_dofs_per_cell();
271
272 FullMatrix<double> cell_matrix(dofs_per_cell, dofs_per_cell);
273 Vector<double> cell_rhs(dofs_per_cell);
274
275 std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
276 Vector<double> transport_values(dim);
277
278 for (const auto &cell : dof_handler.active_cell_iterators())
279 {
280 cell_matrix = 0;
281 cell_rhs = 0;
282
283 fe_values.reinit(cell);
284
285 for (const unsigned int q_index : fe_values.quadrature_point_indices())
286 {
287 const Point<dim> &x_q = fe_values.quadrature_point(q_index);
288 const double diffusion = diffusion_coefficient.value(x_q);
289 const double reaction = reaction_coefficient.value(x_q);
290 const double rhs = forcing_term.value(x_q);
291 const double target = desired_state.value(x_q);
292
293 transport_field.vector_value(x_q, transport_values);
294 Tensor<1, dim> transport;
295 for (unsigned int d = 0; d < dim; ++d)
296 transport[d] = transport_values[d];
297
298 for (const unsigned int i : fe_values.dof_indices())
299 {
300 const double phi_state_i = fe_values[state].value(i, q_index);
301 const Tensor<1, dim> grad_state_i =
302 fe_values[state].gradient(i, q_index);
303 const double phi_adjoint_i = fe_values[adjoint].value(i, q_index);
304 const Tensor<1, dim> grad_adjoint_i =
305 fe_values[adjoint].gradient(i, q_index);
306 const double phi_control_i = fe_values[control].value(i, q_index);
307
308 for (const unsigned int j : fe_values.dof_indices())
309 {
310 const double phi_state_j = fe_values[state].value(j, q_index);
311 const Tensor<1, dim> grad_state_j =
312 fe_values[state].gradient(j, q_index);
313 const double phi_adjoint_j =
314 fe_values[adjoint].value(j, q_index);
315 const Tensor<1, dim> grad_adjoint_j =
316 fe_values[adjoint].gradient(j, q_index);
317 const double phi_control_j =
318 fe_values[control].value(j, q_index);
319
320 // Tracking term on the state variable.
321 cell_matrix(i, j) +=
322 phi_state_i * phi_state_j * fe_values.JxW(q_index);
323
324 // State equation block A(y,p).
325 cell_matrix(i, j) +=
326 (diffusion * grad_state_i * grad_adjoint_j +
327 (transport * grad_state_i) * phi_adjoint_j +
328 reaction * phi_state_i * phi_adjoint_j) *
329 fe_values.JxW(q_index);
330
331 // Adjoint equation block A'(p,y).
332 cell_matrix(i, j) +=
333 (diffusion * grad_adjoint_i * grad_state_j +
334 (transport * grad_state_j) * phi_adjoint_i +
335 reaction * phi_state_j * phi_adjoint_i) *
336 fe_values.JxW(q_index);
337
338 // Optimality block coupling adjoint and control.
339 cell_matrix(i, j) +=
340 (-phi_adjoint_i * phi_control_j -
341 phi_control_i * phi_adjoint_j +
342 regularization * phi_control_i * phi_control_j) *
343 fe_values.JxW(q_index);
344 }
345
346 // Right-hand side for tracking and forcing.
347 cell_rhs(i) += (target * phi_state_i + rhs * phi_adjoint_i) *
348 fe_values.JxW(q_index);
349 }
350 }
351
352 cell->get_dof_indices(local_dof_indices);
353 constraints.distribute_local_to_global(
354 cell_matrix, cell_rhs, local_dof_indices, system_matrix, system_rhs);
355 }
356}

360template <int dim>
361void
362KKT<dim>::output_results(const std::string &filename) const
363{
364 std::vector<std::string> solution_names = {"state", "adjoint", "control"};
365 std::vector<DataComponentInterpretation::DataComponentInterpretation>
366 interpretation(3, DataComponentInterpretation::component_is_scalar);
367 BlockVector<double> desired_state_output;
368 desired_state_output.reinit(
369 std::vector<types::global_dof_index>(dofs_per_block.begin(),
370 dofs_per_block.end()));
371 const VectorFunctionFromScalarFunctionObject<dim> desired_state_function(
372 [this](const Point<dim> &p) { return desired_state.value(p); },
373 state_block,
374 n_blocks);
375
376 VectorTools::interpolate(dof_handler,
377 desired_state_function,
378 desired_state_output);
379
380 DataOut<dim> data_out;
381 data_out.attach_dof_handler(dof_handler);
382 data_out.add_data_vector(dof_handler,
383 solution,
384 solution_names,
385 interpretation);
386 data_out.add_data_vector(dof_handler,
387 desired_state_output,
388 {"desired_state", "zero_adjoint", "zero_control"},
389 interpretation);
390 data_out.build_patches();
391
392 std::ofstream output(filename + ".vtu");
393 data_out.write_vtu(output);
394}
