130template <int dim>
131void
132Laplacian<dim>::setup_system()
133{
134 dof_handler.distribute_dofs(*fe);
135
136 solution.reinit(dof_handler.n_dofs());
137 system_rhs.reinit(dof_handler.n_dofs());
138
139 constraints.clear();
140 DoFTools::make_hanging_node_constraints(dof_handler, constraints);
141 VectorTools::interpolate_boundary_values(
142 dof_handler, 0, boundary_values, constraints);
143 constraints.close();
144
145 DynamicSparsityPattern dsp(dof_handler.n_dofs());
146 DoFTools::make_sparsity_pattern(dof_handler,
147 dsp,
148 constraints,
149 /*keep_constrained_dofs = */ false);
150 sparsity_pattern.copy_from(dsp);
151 system_matrix.reinit(sparsity_pattern);
152 mass_matrix.reinit(sparsity_pattern);
153}

157template <int dim>
158void
159Laplacian<dim>::assemble_system()
160{
161 system_matrix = 0;
162 mass_matrix = 0;
163 system_rhs = 0;
164
165 const QGauss<dim> quadrature_formula(fe->degree + 1);
166 FEValues<dim> fe_values(*fe,
167 quadrature_formula,
168 update_values | update_gradients |
169 update_quadrature_points | update_JxW_values);
170
171 const unsigned int dofs_per_cell = fe->n_dofs_per_cell();
172
173 FullMatrix<double> cell_matrix(dofs_per_cell, dofs_per_cell);
174 FullMatrix<double> cell_mass_matrix(dofs_per_cell, dofs_per_cell);
175 Vector<double> cell_rhs(dofs_per_cell);
176
177 std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
178
179 for (const auto &cell : dof_handler.active_cell_iterators())
180 {
181 cell_matrix = 0;
182 cell_mass_matrix = 0;
183 cell_rhs = 0;
184
185 fe_values.reinit(cell);
186
187 for (const unsigned int q_index : fe_values.quadrature_point_indices())
188 {
189 const double current_coefficient =
190 diffusion_coefficient.value(fe_values.quadrature_point(q_index));
191 const double current_rhs =
192 forcing_term.value(fe_values.quadrature_point(q_index));
193
194 for (const unsigned int i : fe_values.dof_indices())
195 {
196 for (const unsigned int j : fe_values.dof_indices())
197 {
198 cell_matrix(i, j) +=
199 current_coefficient * fe_values.shape_grad(i, q_index) *
200 fe_values.shape_grad(j, q_index) * fe_values.JxW(q_index);
201 cell_mass_matrix(i, j) +=
202 fe_values.shape_value(i, q_index) *
203 fe_values.shape_value(j, q_index) * fe_values.JxW(q_index);
204 }
205
206 cell_rhs(i) += current_rhs * fe_values.shape_value(i, q_index) *
207 fe_values.JxW(q_index);
208 }
209 }
210
211 cell->get_dof_indices(local_dof_indices);
212 constraints.distribute_local_to_global(
213 cell_matrix, cell_rhs, local_dof_indices, system_matrix, system_rhs);
214 constraints.distribute_local_to_global(
215 cell_mass_matrix, local_dof_indices, mass_matrix);
216 }
217}

230template <int dim>
231void
232Laplacian<dim>::solve(const Vector<double> &rhs, Vector<double> &dst) const
233{
234 dst.reinit(solution);
235
236 SolverControl solver_control(solver_max_iterations, solver_tolerance);
237 SolverCG<Vector<double>> solver(solver_control);
238
239 PreconditionSSOR<SparseMatrix<double>> preconditioner;
240 preconditioner.initialize(system_matrix, preconditioner_relaxation);
241
242 solver.solve(system_matrix, dst, rhs, preconditioner);
243 constraints.distribute(dst);
244}
