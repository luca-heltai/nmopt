175 template <typename VectorType>
176 LinearOperator<VectorType>
177 outer_product_operator(const VectorType &left, const VectorType &right)
178 {
179 const auto left_ptr = std::make_shared<VectorType>(left);
180 const auto right_ptr = std::make_shared<VectorType>(right);
181
182 auto op = dealii::identity_operator<VectorType>(
183 [left_ptr](VectorType &v, const bool omit_zeroing_entries) {
184 v.reinit(*left_ptr, omit_zeroing_entries);
185 });
186
187 op.vmult = [left_ptr, right_ptr](VectorType &dst, const VectorType &src) {
188 dst = *left_ptr;
189 dst *= ((*right_ptr) * src);
190 };
191
192 op.vmult_add = [left_ptr, right_ptr](VectorType &dst,
193 const VectorType &src) {
194 dst.add((*right_ptr) * src, *left_ptr);
195 };
196
197 op.Tvmult = [left_ptr, right_ptr](VectorType &dst,
198 const VectorType &src) {
199 dst = *right_ptr;
200 dst *= ((*left_ptr) * src);
201 };
202
203 op.Tvmult_add = [left_ptr, right_ptr](VectorType &dst,
204 const VectorType &src) {
205 dst.add((*left_ptr) * src, *right_ptr);
206 };
207
208 return op;
209 }

273 template <typename VectorType,
274 typename ValueFunction,
275 typename GradientFunction>
276 typename VectorType::value_type
277 armijo_backtracking(
278 const ValueFunction &value,
279 const GradientFunction &gradient,
280 const VectorType &x,
281 const VectorType &p,
282 const ArmijoParameters<typename VectorType::value_type> &parameters = {})
283 {
284 using Number = typename VectorType::value_type;
285
286 const VectorType g = gradient(x);
287 const Number fx = value(x);
288 const Number g_dot_p = g * p;
289
290 Number alpha = parameters.alpha0;
291
292 for (unsigned int backtrack = 0; backtrack < parameters.max_backtracks;
293 ++backtrack)
294 {
295 const VectorType trial = x + alpha * p;
296
297 if (value(trial) <= fx + parameters.c1 * alpha * g_dot_p)
298 return alpha;
299
300 alpha *= parameters.beta;
301 if (alpha <= parameters.minimum_alpha)
302 break;
303 }
304
305 return std::max(alpha, parameters.minimum_alpha);
306 }

312 template <typename VectorType,
313 typename ValueFunction,
314 typename GradientFunction>
315 OptimizationResult<VectorType>
316 optimize_gd(const ValueFunction &value,
317 const GradientFunction &gradient,
318 const VectorType &x0,
319 const OptimizationParameters<typename VectorType::value_type>
320 &parameters = {},
321 const IterationCallback<VectorType> &callback = {})
322 {
323 using Number = typename VectorType::value_type;
324
325 OptimizationResult<VectorType> result;
326 VectorType x = x0;
327
328 internal::maybe_store_iterate(result, x, parameters.store_iterates);
329
330 for (unsigned int k = 0; k < parameters.max_iterations; ++k)
331 {
332 const VectorType g = gradient(x);
333 const Number gnorm = g.l2_norm();
334
335 const Number fx = value(x);
336 result.function_values.push_back(fx);
337 result.gradient_norms.push_back(gnorm);
338 result.iterations = k;
339 internal::maybe_invoke_callback(callback, k, x, fx, gnorm);
340
341 internal::log_iteration(parameters.log_iterations, "GD", k, fx, gnorm);
342
343 if (gnorm < parameters.gradient_tolerance)
344 {
345 internal::log_endline(parameters.log_iterations);
346 break;
347 }
348
349 const VectorType p = Number(-1.) * g;
350 const Number alpha =
351 armijo_backtracking(value, gradient, x, p, parameters.armijo);
352
353 x = x + alpha * p;
354
355 result.step_lengths.push_back(alpha);
356 internal::log_scalar(parameters.log_iterations, "alpha", alpha);
357 internal::log_endline(parameters.log_iterations);
358 internal::maybe_store_iterate(result, x, parameters.store_iterates);
359 }
360
361 result.x = x;
362 return result;
363 }

369 template <typename VectorType,
370 typename ValueFunction,
371 typename GradientFunction>
372 OptimizationResult<VectorType>
373 optimize_nlcg(
374 const ValueFunction &value,
375 const GradientFunction &gradient,
376 const VectorType &x0,
377 const NLCGParameters<typename VectorType::value_type> &parameters = {},
378 const IterationCallback<VectorType> &callback = {})
379 {
380 using Number = typename VectorType::value_type;
381
382 OptimizationResult<VectorType> result;
383 VectorType x = x0;
384 VectorType g = gradient(x);
385 VectorType p = Number(-1.) * g;
386
387 internal::maybe_store_iterate(result, x, parameters.store_iterates);
388
389 for (unsigned int k = 0; k < parameters.max_iterations; ++k)
390 {
391 const Number gnorm = g.l2_norm();
392
393 const Number fx = value(x);
394 result.function_values.push_back(fx);
395 result.gradient_norms.push_back(gnorm);
396 result.iterations = k;
397 internal::maybe_invoke_callback(callback, k, x, fx, gnorm);
398
399 internal::log_iteration(
400 parameters.log_iterations, "NLCG", k, fx, gnorm);
401
402 if (gnorm < parameters.gradient_tolerance)
403 {
404 internal::log_endline(parameters.log_iterations);
405 break;
406 }
407
408 const Number alpha =
409 armijo_backtracking(value, gradient, x, p, parameters.armijo);
410
411 const VectorType x_new = x + alpha * p;
412 const VectorType g_new = gradient(x_new);
413
414 Number beta = Number();
415
416 const Number gg =
417 std::max(g * g, std::numeric_limits<Number>::epsilon());
418
419 if (parameters.beta_type == "fletcher_reeves")
420 beta = (g_new * g_new) / gg;
421 else if (parameters.beta_type == "polak_ribiere_plus")
422 {
423 const VectorType y = g_new - g;
424 beta = std::max(Number(), (g_new * y) / gg);
425 }
426 else
427 AssertThrow(false,
428 ExcMessage("Unknown NLCG beta type: " +
429 parameters.beta_type));
430
431 if ((k + 1) % parameters.restart_every == 0)
432 beta = Number();
433
434 p = Number(-1.) * g_new + beta * p;
435
436 if ((p * g_new) >= Number())
437 p = Number(-1.) * g_new;
438
439 x = x_new;
440 g = g_new;
441
442 result.step_lengths.push_back(alpha);
443 internal::log_scalar(parameters.log_iterations, "alpha", alpha);
444 internal::log_scalar(parameters.log_iterations, "beta", beta);
445 internal::log_endline(parameters.log_iterations);
446 internal::maybe_store_iterate(result, x, parameters.store_iterates);
447 }
448
449 result.x = x;
450 return result;
451 }

457 template <typename VectorType,
458 typename ValueFunction,
459 typename GradientFunction>
460 OptimizationResult<VectorType>
461 optimize_bfgs(
462 const ValueFunction &value,
463 const GradientFunction &gradient,
464 const VectorType &x0,
465 const LBFGSParameters<typename VectorType::value_type> &parameters = {},
466 const IterationCallback<VectorType> &callback = {})
467 {
468 using Number = typename VectorType::value_type;
469
470 OptimizationResult<VectorType> result;
471 VectorType x = x0;
472 std::vector<VectorType> s_history;
473 std::vector<VectorType> y_history;
474 std::vector<Number> rho_history;
475
476 internal::maybe_store_iterate(result, x, parameters.store_iterates);
477
478 for (unsigned int k = 0; k < parameters.max_iterations; ++k)
479 {
480 const VectorType g = gradient(x);
481 const Number gnorm = g.l2_norm();
482
483 const Number fx = value(x);
484 result.function_values.push_back(fx);
485 result.gradient_norms.push_back(gnorm);
486 result.iterations = k;
487 internal::maybe_invoke_callback(callback, k, x, fx, gnorm);
488
489 internal::log_iteration(
490 parameters.log_iterations, "LBFGS", k, fx, gnorm);
491
492 if (gnorm < parameters.gradient_tolerance)
493 {
494 internal::log_endline(parameters.log_iterations);
495 break;
496 }
497
498 VectorType q = g;
499 const unsigned int m = s_history.size();
500 std::vector<Number> alphas(m);
501
502 for (int i = static_cast<int>(m) - 1; i >= 0; --i)
503 {
504 alphas[i] = rho_history[i] * (s_history[i] * q);
505 q -= alphas[i] * y_history[i];
506 }
507
508 Number gamma = Number(1.);
509 if (m > 0)
510 {
511 const Number yy = y_history.back() * y_history.back();
512 if (yy > Number(0.))
513 gamma = (s_history.back() * y_history.back()) / yy;
514 }
515
516 VectorType r = gamma * q;
517
518 for (unsigned int i = 0; i < m; ++i)
519 {
520 const Number beta = rho_history[i] * (y_history[i] * r);
521 r += (alphas[i] - beta) * s_history[i];
522 }
523
524 VectorType p = Number(-1.) * r;
525
526 if ((p * g) >= Number())
527 {
528 p = Number(-1.) * g;
529 s_history.clear();
530 y_history.clear();
531 rho_history.clear();
532 }
533
534 const Number alpha =
535 armijo_backtracking(value, gradient, x, p, parameters.armijo);
536
537 const VectorType s = alpha * p;
538 const VectorType x_new = x + s;
539 const VectorType y = gradient(x_new) - g;
540 const Number ys = y * s;
541
542 if (ys > Number(1.e-12))
543 {
544 if (s_history.size() == parameters.history_size)
545 {
546 s_history.erase(s_history.begin());
547 y_history.erase(y_history.begin());
548 rho_history.erase(rho_history.begin());
549 }
550
551 s_history.push_back(s);
552 y_history.push_back(y);
553 rho_history.push_back(Number(1.) / ys);
554 }
555 else
556 {
557 s_history.clear();
558 y_history.clear();
559 rho_history.clear();
560 }
561
562 x = x_new;
563
564 result.step_lengths.push_back(alpha);
565 internal::log_scalar(parameters.log_iterations, "alpha", alpha);
566 internal::log_scalar(parameters.log_iterations,
567 "history",
568 s_history.size());
569 internal::log_endline(parameters.log_iterations);
570 internal::maybe_store_iterate(result, x, parameters.store_iterates);
571 }
572
573 result.x = x;
574 return result;
575 }

583 template <typename VectorType,
584 typename ValueFunction,
585 typename GradientFunction,
586 typename HessianFunction>
587 OptimizationResult<VectorType>
588 optimize_trust_region_cauchy(
589 const ValueFunction &value,
590 const GradientFunction &gradient,
591 const HessianFunction &hessian,
592 const VectorType &x0,
593 const TrustRegionParameters<typename VectorType::value_type> &parameters =
594 {},
595 const IterationCallback<VectorType> &callback = {})
596 {
597 using Number = typename VectorType::value_type;
598
599 OptimizationResult<VectorType> result;
600 VectorType x = x0;
601 Number delta = parameters.delta0;
602
603 internal::maybe_store_iterate(result, x, parameters.store_iterates);
604
605 for (unsigned int k = 0; k < parameters.max_iterations; ++k)
606 {
607 const VectorType g = gradient(x);
608 const Number gnorm = g.l2_norm();
609 const auto B = hessian(x);
610
611 const Number fx = value(x);
612 result.function_values.push_back(fx);
613 result.gradient_norms.push_back(gnorm);
614 result.trust_region_radii.push_back(delta);
615 result.iterations = k;
616 internal::maybe_invoke_callback(callback, k, x, fx, gnorm);
617
618 internal::log_iteration(parameters.log_iterations, "TR", k, fx, gnorm);
619 internal::log_scalar(parameters.log_iterations, "delta", delta);
620
621 if (gnorm < parameters.gradient_tolerance)
622 {
623 internal::log_endline(parameters.log_iterations);
624 break;
625 }
626
627 const VectorType Bg = B * g;
628 const Number gBg = g * Bg;
629
630 Number tau = Number(1.);
631 if (gBg > Number())
632 tau = std::min((gnorm * gnorm * gnorm) / (delta * gBg), Number(1.));
633
634 const VectorType p = (-tau * delta / gnorm) * g;
635
636 const Number ared = value(x) - value(x + p);
637 const Number pred = -(g * p + Number(0.5) * (p * (B * p)));
638 const Number rho =
639 ared / std::max(pred, std::numeric_limits<Number>::epsilon());
640
641 if (rho < Number(0.25))
642 delta *= Number(0.25);
643 else if (rho > Number(0.75) &&
644 std::abs(p.l2_norm() - delta) < Number(1.e-12))
645 delta = std::min(Number(2.) * delta, parameters.delta_max);
646
647 if (rho > parameters.eta)
648 {
649 x = x + p;
650 result.step_lengths.push_back(p.l2_norm());
651 internal::maybe_store_iterate(result, x, parameters.store_iterates);
652 }
653
654 internal::log_scalar(parameters.log_iterations, "rho", rho);
655 internal::log_endline(parameters.log_iterations);
656 }
657
658 result.x = x;
659 return result;
660 }
